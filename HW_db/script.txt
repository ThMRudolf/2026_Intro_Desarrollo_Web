-- ============================================
-- DROPS (solo si existen)
-- ============================================
DROP TABLE IF EXISTS Reproducciones;
DROP TABLE IF EXISTS PeliculaActor;
DROP TABLE IF EXISTS Usuarios;
DROP TABLE IF EXISTS Peliculas;
DROP TABLE IF EXISTS Actores;
DROP TABLE IF EXISTS Generos;
DROP TABLE IF EXISTS Paises;

-- ============================================
-- CREATE TABLES
-- ============================================

CREATE TABLE Paises (
    id INTEGER PRIMARY KEY,
    nombre TEXT NOT NULL
);

CREATE TABLE Generos (
    id INTEGER PRIMARY KEY,
    nombre TEXT NOT NULL
);

CREATE TABLE Actores (
    id INTEGER PRIMARY KEY,
    nombre TEXT NOT NULL
);

CREATE TABLE Peliculas (
    id INTEGER PRIMARY KEY,
    titulo TEXT NOT NULL,
    genero_id INTEGER NOT NULL,
    año INTEGER,
    rating_promedio REAL,
    FOREIGN KEY (genero_id) REFERENCES Generos(id)
);

CREATE TABLE PeliculaActor (
    id INTEGER PRIMARY KEY,
    pelicula_id INTEGER NOT NULL,
    actor_id INTEGER NOT NULL,
    FOREIGN KEY (pelicula_id) REFERENCES Peliculas(id),
    FOREIGN KEY (actor_id) REFERENCES Actores(id)
);

CREATE TABLE Usuarios (
    id INTEGER PRIMARY KEY,
    nombre TEXT NOT NULL,
    pais_id INTEGER NOT NULL,
    fecha_registro TEXT,
    FOREIGN KEY (pais_id) REFERENCES Paises(id)
);

CREATE TABLE Reproducciones (
    id INTEGER PRIMARY KEY,
    usuario_id INTEGER NOT NULL,
    pelicula_id INTEGER NOT NULL,
    fecha TEXT,
    minutos_vistos INTEGER,
    FOREIGN KEY (usuario_id) REFERENCES Usuarios(id),
    FOREIGN KEY (pelicula_id) REFERENCES Peliculas(id)
);
